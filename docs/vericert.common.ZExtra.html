Module ZExtra.

  Lemma mod_0_bounds :
    forall x y m,
      0 < m ->
      x mod m = 0 ->
      y mod m = 0 ->
      x <> y ->
      x + m > y ->
      y + m <= x.

  Lemma Ple_not_eq :
    forall x y,
    (x < y)%positive -> x <> y.

  Lemma Pge_not_eq :
    forall x y,
    (y < x)%positive -> x <> y.

  Lemma Ple_Plt_Succ :
    forall x y n,
    (x <= y)%positive -> (x < y + n)%positive.

End ZExtra.
